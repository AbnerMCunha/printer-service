# ============================================
# Configuração do Serviço de Impressão
# ============================================
# Copie este arquivo para .env e preencha com seus valores

# URL da API do backend
# Exemplo: https://api.cardapix.net
API_URL=https://api.cardapix.net

# Token JWT do admin para autenticação (Access Token)
# Obtenha este token executando: npm run get-token
API_TOKEN=seu-token-jwt-aqui

# Refresh Token para renovação automática (recomendado)
# Configure para renovação automática de tokens - funciona 24/7 sem intervenção
# Obtido junto com o API_TOKEN ao executar: npm run get-token
REFRESH_TOKEN=seu-refresh-token-aqui

# Credenciais para login automático (opcional, mas recomendado)
# Quando refresh token expirar, o serviço fará login automático usando estas credenciais
# Configure para ZERO manutenção de tokens
ADMIN_EMAIL=admin@restaurant.com
ADMIN_PASSWORD=sua-senha-aqui

# ID do restaurante (opcional)
RESTAURANT_ID=

# Tipo de impressora: 'thermal' (térmica via TCP/IP) ou 'system' (normal via spooler)
# Padrão: thermal
PRINTER_TYPE=thermal

# Nome da impressora no sistema (apenas para PRINTER_TYPE=system)
# No Windows: nome exato da impressora nas configurações do Windows
# No Linux: nome retornado por 'lpstat -p'
# Deixe vazio para usar impressora padrão do sistema
PRINTER_NAME=

# IP da impressora térmica na rede (apenas para PRINTER_TYPE=thermal)
PRINTER_IP=192.168.1.100

# Porta da impressora térmica (apenas para PRINTER_TYPE=thermal, padrão: 9100)
PRINTER_PORT=9100

# Porta do servidor HTTP local (padrão: 3002)
# O frontend enviará requisições para esta porta quando detectar novos pedidos
HTTP_PORT=3002

# Habilitar polling como fallback (padrão: true)
# Se false, o serviço só responderá a requisições HTTP do frontend
# Se true, fará polling mesmo quando frontend não estiver aberto
ENABLE_POLLING=true

# Intervalo de polling em milissegundos (padrão: 30000)
# Usado apenas quando ENABLE_POLLING=true
POLLING_INTERVAL=30000

# Nível de log (error, warn, info, debug)
LOG_LEVEL=info
